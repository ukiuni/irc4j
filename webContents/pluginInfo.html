指定したチャネルにメッセージを送信します。
"/COMMAND channelName message" というフォーマットを期待した実装です。 
function execute(server, client, params) {
    var channelName = params[0];
    var channel = server.getChannel(channelName);
    var message = params[1];
    client.sendMessage("PRIVMSG", client, channel, message);
}
10秒後にサーバーメッセージを返信します。 
function execute(server, client, params) {
    setTimeout(function(){
        client.sendPrivateCommand("Hello "+client.getNickName()+"!!");
    }, 10000);
}
